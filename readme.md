# Clone the repo
$ git clone https://github.com/iamsingh-h/News-Today/new/main

# Go to client 
$ cd client

# Install the dependencies 
$npm i

# Go to server 
$ cd client

# Install the dependencies 
$npm i

# Start the client
npm start

# Start the server
npm run dev

# Open Browser
Open http://localhost:3000 to view it in the browser.
